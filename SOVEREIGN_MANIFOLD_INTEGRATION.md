# council/tests/geometry/test_mathematical_properties.py

import pytest
import torch
from council.core.geometry import *

class TestManifoldProperties:

    def test_geodesics_are_shortest_paths(self):
        """Geodesics must be locally shortest paths"""
        manifold = Sphere(n=10)
        p1 = manifold.random_point()
        p2 = manifold.random_point()

        # Compute geodesic
        geodesic = manifold.geodesic(p1, p2)
        geodesic_length = manifold.length(geodesic)

        # Compare with perturbed paths
        for _ in range(100):
            perturbed = manifold.perturb_path(geodesic)
            perturbed_length = manifold.length(perturbed)
            assert geodesic_length <= perturbed_length

    def test_stiefel_frames_orthonormal(self):
        """Stiefel frames must be orthonormal"""
        stiefel = Stiefel(n=50, p=475)
        frame = stiefel.random_point()

        # Check orthonormality: X^T X = I
        gram = frame.T @ frame
        identity = torch.eye(50)
        assert torch.allclose(gram, identity, atol=1e-6)

    def test_hyperbolic_tree_embedding(self):
        """Tree distances must be preserved in hyperbolic space"""
        tree = generate_binary_tree(depth=10)  # 1024 nodes
        hyperbolic = HyperbolicMemory(dim=384)

        embeddings = hyperbolic.embed_tree(tree)

        # Check all pairwise distances
        for (u, v), tree_dist in tree.distances.items():
            emb_u = embeddings[u]
            emb_v = embeddings[v]
            hyp_dist = hyperbolic.distance(emb_u, emb_v)

            # Allow 5% distortion
            assert abs(hyp_dist - tree_dist) / tree_dist < 0.05

    def test_pareto_front_validity(self):
        """All Pareto front solutions must be non-dominated"""
        optimizer = ParetoOptimizer()
        objectives = [
            lambda x: x[0]**2,      # Minimize
            lambda x: -(x[1]**2)    # Maximize (negate for minimization)
        ]

        pareto_front = optimizer.optimize(objectives, manifold=Sphere(n=2))

        # Check: no solution dominates another
        for i, sol_i in enumerate(pareto_front):
            for j, sol_j in enumerate(pareto_front):
                if i != j:
                    assert not dominates(sol_i, sol_j)
